    updateMainList();
  }

  function downloadSurah(surah) {
    const a = document.createElement('a');
    a.href = surah.src;
    a.download = `${surah.number}_${surah.nameAr}.mp3`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    showMessage(`⏬ تحميل ${surah.nameAr}`);
  }

  function createSurahElement(surah, idxGlobal) {
    const isFav = favorites.some(f => f.number === surah.number);
    const div = document.createElement('div');
    div.className = 'surah-item';
    div.dataset.index = surah.number-1;
    div.innerHTML = `
      <div class="surah-info">
        <div class="surah-number">${surah.number}</div>
        <div class="surah-details">
          <div class="surah-name-ar">${surah.nameAr}</div>
          <div class="surah-name">${surah.name}</div>
        </div>
      </div>
      <div class="surah-controls">
        <button class="play-btn" title="تشغيل"><i class="fas fa-play"></i></button>
        <button class="download-btn" title="تحميل"><i class="fas fa-download"></i></button>
        <button class="favorite-btn ${isFav ? 'active' : ''}" title="مفضلة"><i class="fas fa-heart"></i></button>
      </div>
    `;
    const playBtn = div.querySelector('.play-btn');
    const downBtn = div.querySelector('.download-btn');
    const favBtn = div.querySelector('.favorite-btn');
    
    playBtn.onclick = (e) => {
      e.stopPropagation();
      const idxReal = surah.number-1;
      if(currentIndex === idxReal && isPlaying) pauseSurah();
      else { currentIndex = idxReal; loadSurah(currentIndex); playSurah(); }
    };
    downBtn.onclick = (e) => { e.stopPropagation(); downloadSurah(surah); };
    favBtn.onclick = (e) => {
      e.stopPropagation();
      const idxFav = favorites.findIndex(f => f.number === surah.number);
      if(idxFav === -1) {
        favorites.push({ number: surah.number, nameAr: surah.nameAr, name: surah.name });
        showMessage(`❤️ أضيفت ${surah.nameAr} إلى المفضلة`);
      } else {
        favorites.splice(idxFav,1);
        showMessage(`♡ أزيلت ${surah.nameAr} من المفضلة`);
      }
      localStorage.setItem('quranFav', JSON.stringify(favorites));
      favBtn.classList.toggle('active');
      updateFavoritesPage();
      updateMainList();
    };
    div.onclick = () => {
      const idxReal = surah.number-1;
      if(currentIndex === idxReal && isPlaying) pauseSurah();
      else { currentIndex = idxReal; loadSurah(currentIndex); playSurah(); }
    };
    return div;
  }

  function updateMainList() {
    const container = document.getElementById('surah-list');
    container.innerHTML = '';
    filteredSurahs.forEach(s => container.appendChild(createSurahElement(s, s.number-1)));
    updatePlayingState();
  }
  
  function updateFavoritesPage() {
    const container = document.getElementById('favorites-list');
    if(favorites.length === 0) {
      container.innerHTML = '<div class="empty-message"><i class="far fa-heart"></i>  لا توجد سور مفضلة بعد<br>أضف من القائمة الرئيسية ♡</div>';
      return;
    }
    container.innerHTML = '';
    favorites.forEach(fav => {
      const original = surahsData.find(s => s.number === fav.number);
      if(original) container.appendChild(createSurahElement(original, original.number-1));
    });
  }
  
  function updatePlayingState() {
    document.querySelectorAll('.surah-item').forEach(item => {
      const idx = parseInt(item.dataset.index);
      if(idx === currentIndex && isPlaying) {
        item.classList.add('playing');
        const icon = item.querySelector('.play-btn i');
        if(icon) icon.className = 'fas fa-pause';
      } else {
        item.classList.remove('playing');
        const icon = item.querySelector('.play-btn i');
        if(icon) icon.className = 'fas fa-play';
      }
    });
  }
  
  function loadSurah(index) {
    const surah = surahsData[index];
    if(!surah) return;
    audio.src = surah.src;
    currentTitleSpan.innerHTML = `<i class="fas fa-headphones"></i> ${surah.number}. ${surah.nameAr} - ${recitersNames[currentReciter]}`;
    durationSpan.textContent = '⏳';
    progressFill.style.width = '0%';
    currentTimeSpan.textContent = '0:00';
    localStorage.setItem('lastSurah', index);
    updatePlayingState();
    audio.onloadedmetadata = () => { if(audio.duration) durationSpan.textContent = formatTime(audio.duration); };
  }
  
  function formatTime(secs) {
    if(isNaN(secs)) return "0:00";
    const mins = Math.floor(secs / 60);
    const remainSec = Math.floor(secs % 60);
    return `${mins}:${remainSec.toString().padStart(2,'0')}`;
  }
  
  function playSurah() {
    if(!audio.src || audio.src === window.location.href) loadSurah(currentIndex);
    audio.play().then(() => {
      playGlobalBtn.innerHTML = '<i class="fas fa-pause"></i>';
      isPlaying = true;
      updatePlayingState();
    }).catch(e => { showMessage("⚠️ تعذر التشغيل، تأكد من الاتصال", true); isPlaying = false; playGlobalBtn.innerHTML = '<i class="fas fa-play"></i>'; });
  }
  
  function pauseSurah() { audio.pause(); playGlobalBtn.innerHTML = '<i class="fas fa-play"></i>'; isPlaying = false; updatePlayingState(); }
  function togglePlay() { isPlaying ? pauseSurah() : playSurah(); }
  function nextSurah() { currentIndex = (currentIndex + 1) % surahsData.length; loadSurah(currentIndex); if(isPlaying) playSurah(); else updatePlayingState(); }
  function prevSurah() { currentIndex = (currentIndex - 1 + surahsData.length) % surahsData.length; loadSurah(currentIndex); if(isPlaying) playSurah(); else updatePlayingState(); }
  
  function onEnded() { if(isRepeat) { audio.currentTime = 0; playSurah(); } else nextSurah(); }
  
  function updateProgressUI() {
    if(audio.duration && !isNaN(audio.duration)) {
      const percent = (audio.currentTime / audio.duration) * 100;
      progressFill.style.width = `${percent}%`;
      currentTimeSpan.textContent = formatTime(audio.currentTime);
    }
  }
  function setProgress(e) { const rect = progressBar.getBoundingClientRect(); const percent = (e.clientX - rect.left)/rect.width; if(audio.duration) audio.currentTime = percent * audio.duration; }
  function updateVolume() { audio.volume = volumeSlider.value/100; volumeValue.textContent = `${volumeSlider.value}%`; localStorage.setItem('quranVol', volumeSlider.value); }
  function volUp() { volumeSlider.value = Math.min(100, parseInt(volumeSlider.value)+10); updateVolume(); }
  function volDown() { volumeSlider.value = Math.max(0, parseInt(volumeSlider.value)-10); updateVolume(); }
  
  function switchPage(pageId) {
    document.querySelectorAll('.content-section').forEach(s => s.classList.remove('active'));
    document.getElementById(`${pageId}-page`).classList.add('active');
    document.querySelectorAll('.sidebar a').forEach(a => a.classList.remove('active'));
    document.querySelector(`.sidebar a[data-page="${pageId}"]`).classList.add('active');
    if(pageId === 'favorites') updateFavoritesPage();
  }
  
  function changeReciter(recId) {
    if(currentReciter === recId) return;
    currentReciter = recId;
    currentBaseUrl = recitersUrls[recId];
    document.querySelectorAll('.reciter-btn').forEach(btn => btn.classList.remove('active'));
    document.querySelector(`.reciter-btn[data-reciter="${recId}"]`).classList.add('active');
    updateReciterUrls();
  }
  
  function loadSavedSettings() {
    const last = localStorage.getItem('lastSurah');
    if(last && surahsData[last]) currentIndex = parseInt(last);
    const vol = localStorage.getItem('quranVol');
    if(vol) { volumeSlider.value = vol; updateVolume(); }
    loadSurah(currentIndex);
  }
  
  // Events
  playGlobalBtn.onclick = togglePlay;
  nextGlobalBtn.onclick = nextSurah;
  prevGlobalBtn.onclick = prevSurah;
  repeatGlobalBtn.onclick = () => { isRepeat = !isRepeat; repeatGlobalBtn.style.background = isRepeat ? '#ffd700' : 'rgba(255,255,255,0.2)'; repeatGlobalBtn.style.color = isRepeat ? '#0a2f1f' : 'white'; showMessage(isRepeat ? "🔁 تكرار السورة نشط" : "🔁 إلغاء التكرار"); };
  progressBar.onclick = setProgress;
  audio.ontimeupdate = updateProgressUI;
  audio.onended = onEnded;
  volumeSlider.oninput = updateVolume;
  volumeUp.onclick = volUp;
  volumeDown.onclick = volDown;
  searchInput.oninput = (e) => filterSurahs(e.target.value);
  document.querySelectorAll('.sidebar a').forEach(a => a.onclick = () => switchPage(a.dataset.page));
  document.querySelectorAll('.reciter-btn').forEach(btn => btn.onclick = () => changeReciter(btn.dataset.reciter));
  
  window.onkeydown = (e) => {
    if(e.code === 'Space') { e.preventDefault(); togglePlay(); }
    if(e.code === 'ArrowRight') { e.preventDefault(); nextSurah(); }
    if(e.code === 'ArrowLeft') { e.preventDefault(); prevSurah(); }
    if(e.code === 'KeyR') { e.preventDefault(); repeatGlobalBtn.click(); }
    if(e.code === 'ArrowUp') { e.preventDefault(); volUp(); }
    if(e.code === 'ArrowDown') { e.preventDefault(); volDown(); }
  };
  
  updateMainList();
  loadSavedSettings();
  updateFavoritesPage();
</script>
</body>
</html>
